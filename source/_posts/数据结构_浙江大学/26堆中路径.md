# def main():

#     读取n m 结点个数，查找目标个数

#     根据输入建立堆(最小堆吧)

#     打印到根的路径


addr = [0]    #利用可变对象的特性。
def Insert(heap,item):
    if heap[0] >= item:
        print('小于哨兵！')
        return
    if addr[0] == 0:
        heap[1] = item # 插入
        addr[0] += 1
        return
    i = addr[0] + 1
    while heap[i//2] >= item: # 找插入位置i。
        heap[i] = heap[i//2]
        i //= 2
    heap[i] = item  # 插入
    addr[0] += 1

def main(H,MinValue=0,list_=[]):
    [n,m] = H[0]
    heap = [None] * (n+1)
    heap[0] = MinValue  # 哨兵
    for i in H[1]:
        Insert(heap,i) # 建堆
    for i in H[2]:     # 打印路径
        while i >= 1:
            list_.append(heap[i])
            i //= 2
        list_.append(' --- ')
    print(list_)
    return list_
if __name__ == '__main__':
    h = [[5, 3],
        [46 ,23 ,26 ,24 ,10],
        [5, 4, 3]]
    main(h)
